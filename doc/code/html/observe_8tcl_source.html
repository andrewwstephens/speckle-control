13 ## Documented proc \c abortsequence .
14 #
15 # This procedure aborts the current exposure or sequence of exposures.
16 # It simply sets the global abort flag and resets the GUI widgets
17 #
18 # Globals :
19 #
20 # STATUS - Exposure status
21 #
22 proc abortsequence { } {
23 global STATUS
24  set STATUS(abort) 1
25  andorSetControl 0 abort
26  .main.observe configure -text "Observe" -bg gray -relief raised -command startsequence
27  .main.abort configure -bg gray -relief sunken -fg LightGray
28  mimicMode red close
29  mimicMode blue close
30 }

33 ## Documented proc \c abortsequence .
34 # \param[in] op - Operation specifier
35 # \param[in] id - Camera id (for multi-camera use) (optional, default is 0)
36 #
37 # This procedure aborts the current exposure or sequence of exposures.
38 # It simply sets the global abort flag and resets the GUI widgets
39 #
40 # Globals :
41 # SCOPE - Telescope parameters, gui setup

43 proc observe { op {id 0} } {
44 global SCOPE
45  switch $op {
46  region128 {acquisitionmode 128}
47  region256 {acquisitionmode 256}
48  region512 {acquisitionmode 512}
49  regionall {acquisitionmode 1024}
50  manual {acquisitionmode manual}
51  multiple {continuousmode $SCOPE(exposure) 999999 $id}
52  fullframe {setfullframe}
53  }
54 }

118  }
119  if { $rdim == "manual" } {
120  set rdim $ACQREGION(geom)
121  set it [tk_dialog .d "Edit regions" "Move the regions in the\n image display tool then click OK" {} -1 "OK"]
122 # commandAndor red "forceroi $ACQREGION(xs) $ACQREGION(xe) $ACQREGION(ys) $ACQREGION(ye)"
123 # commandAndor blue "forceroi $ACQREGION(xs) $ACQREGION(xe) $ACQREGION(ys) $ACQREGION(ye)"
124  } else {
125  set resr [commandAndor red "setroi $rdim"]

134  mimicMode blue roi [set rdim]x[set rdim]
135  exec xpaset -p ds9red regions system physical
136  exec xpaset -p ds9blue regions system physical
137  if { $rdim != 1024 } {
138  set reg [split [exec xpaget ds9red regions] \n]
139  foreach i $reg {
140  if { [string range $i 0 8] == "image;box" || [string range $i 0 2] == "box" } {

142  set ACQREGION(rxs) [expr int([lindex $r 0] - [lindex $r 2]/2)]
143  set ACQREGION(rys) [expr int([lindex $r 1] - [lindex $r 3]/2)]

152  set r [lrange [split $i ",()"] 1 4]
153  set ACQREGION(bxs) [expr int([lindex $r 0] - [lindex $r 2]/2)]
154  set ACQREGION(bys) [expr int([lindex $r 1] - [lindex $r 3]/2)]
155  set ACQREGION(bxe) [expr $ACQREGION(bxs) + [lindex $r 2] -1]
156  set ACQREGION(bye) [expr $ACQREGION(bys) + [lindex $r 3] -1]
157  puts stdout "selected red region $r"
158  }
159  }
160  set CONFIG(geometry.StartCol) [expr $ACQREGION(rxs)]
161  set CONFIG(geometry.StartRow) [expr $ACQREGION(rys)]
162  set CONFIG(geometry.NumCols) $rdim
163  set CONFIG(geometry.NumRows) $rdim
164  set ACQREGION(geom) $CONFIG(geometry.NumCols)
165  debuglog "ROI's are red = $ACQREGION(rxs) $ACQREGION(rys) $ACQREGION(rxe) $ACQREGION(rye)"

167  commandAndor red "setframe roi"
168  commandAndor blue "setframe roi"
169  set LASTACQ roi
170  .lowlevel.rmode configure -text "Mode=ROI"
171  .lowlevel.bmode configure -text "Mode=ROI"
172  } else {
173  set ACQREGION(geom) 1024
174  set ACQREGION(rxs) 1
175  set ACQREGION(rxe) 1024
176  set ACQREGION(rys) 1
177  set ACQREGION(rye) 1024
178  set ACQREGION(bxs) 1
179  set ACQREGION(bxe) 1024
180  set ACQREGION(bys) 1
181  set ACQREGION(bye) 1024
182  if { $ANDOR_CFG(kineticMode) } {
183  commandAndor red "setframe fullkinetic"
184  commandAndor blue "setframe fullkinetic"
185  .lowlevel.rmode configure -text "Mode=FULL"
186  .lowlevel.bmode configure -text "Mode=FULL"

225 ## Documented proc \c startsequence .
226 #
227 # This routine manages a sequence of exposures. It updates bias columns\n
228 # specifications in case they have been changed, then it loops thru\n
229 # a set of frames, updating the progress bar.
230 #
231 # Globals :
232 # SCOPE - Telescope parameters, gui setup
233 # CONFIG - Image geometry configuration
234 # LASTACQ - Type of last image acqusition
235 # ANDOR_DEF - Andor defaults
236 # ACQREGION - Sub-frame region coordinates
237 # ANDOR_CFG - Array of camera settings
238 # ANDOR_SOCKET - Andor camera server socket handles
239 # SCOPE - Telescope parameters, gui setup
240 # OBSPARS - Default observation parameters
241 # FRAME - Frame number in a sequence
242 # STATUS - Exposure status
243 # DEBUG - Set to 1 for verbose logging
244 # TELEMETRY - Array of telemetry for headers and database usage
245 # DATAQUAL - Array of data quality information
246 # INSTRUMENT - Array of instrument configuration data
247 #

259  speckleshutter blue auto

261  commandAndor blue "frametransfer $ANDOR_CFG(blue,frametransfer)"
262  commandAndor red "numberkinetics $SCOPE(numframes)"
263  commandAndor blue "numberkinetics $SCOPE(numframes)"
264  commandAndor red "numberaccumulations $SCOPE(numaccum)"
265  commandAndor blue "numberaccumulations $SCOPE(numaccum)"
266  commandAndor red "programid $SCOPE(ProgID)"
267  commandAndor blue "programid $SCOPE(ProgID)"
268  commandAndor red "autofitds9 $INSTRUMENT(red,fitds9)"
269  commandAndor blue "autofitds9 $INSTRUMENT(blue,fitds9)"

286  } else {
287  commandAndor blue "outputamp $ANDOR_CCD"
288  }
289  commandAndor red "dqtelemetry $DATAQUAL(rawiq) $DATAQUAL(rawcc) $DATAQUAL(rawwv) $DATAQUAL(rawbg)"
290  commandAndor blue "dqtelemetry $DATAQUAL(rawiq) $DATAQUAL(rawcc) $DATAQUAL(rawwv) $DATAQUAL(rawbg)"
291  commandAndor red "filter $SPECKLE_FILTER(red,current)"
292  commandAndor blue "filter $SPECKLE_FILTER(blue,current)"
293  set cmt [join [split [string trim [.main.comment get 0.0 end]] \n] "|"]
294  commandAndor red "comments $cmt"

298  andorSetControl 0 frame 0

306  set OBSPARS($SCOPE(exptype)) "$SCOPE(exposure) $SCOPE(numframes) $SCOPE(shutter)"
307  set STATUS(abort) 0

309  .main.abort configure -bg orange -relief raised -fg black

352  while { $i < $SCOPE(numframes) && $STATUS(abort) == 0 } {

355  if { $DEBUG} {debuglog "$SCOPE(exptype) frame $i"}
356 # after [expr int($perframe*1000)]

358  if { $LASTACQ == "fullframe" } {
359  incr i 1
360  } else {
361  set i [andorGetControl 0 frame]
362  }
363  .lowlevel.p configure -value [expr $i*100/$SCOPE(numframes)]
364  .lowlevel.progress configure -text "Observation status : Frame $i Exposure $dfrmnum Sequence $iseqnum"
