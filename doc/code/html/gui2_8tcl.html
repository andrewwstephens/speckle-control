plotTimings

Documented proc plotTimings .

Plot the timing infornation stored in a binary FITS extension
of an image cube from a kinetic series observation.
The times are stored as TAI, but we subtract the initial time
before plotting. Gnuplot is used to plot.

48 proc plotTimings { } {
49 global env SCOPE
50  set it [tk_getOpenFile -initialdir $SCOPE(datadir)]
51  if { [file exists $it] } {
52  set fh [fits open $it]
53  $fh move +1
54  set times [$fh get table]
55  set fout [open /tmp/timings w]
56  set start [lindex $times 0]
57  foreach i $times {
58  set t [expr $i-$start]
59  if { [expr abs($t)] < 1000 } {puts $fout $t}
60  }
61  close $fout
62  fits close $fh
63  exec echo "plot \"/tmp/timings\"" | gnuplot -p
64  }

123 proc validInteger {win event X oldX min max} {
124  # Make sure min<=max
125  if {$min > $max} {
126  set tmp $min; set min $max; set max $tmp
127  }
128  # Allow valid integers, empty strings, sign without number
129  # Reject Octal numbers, but allow a single "0"
130  # Which signes are allowed ?
131  if {($min <= 0) && ($max >= 0)} { ;# positive & negative sign
132  set pattern {^[+-]?(()|0|([1-9][0-9]*))$}
133  } elseif {$max < 0} { ;# negative sign
134  set pattern {^[-]?(()|0|([1-9][0-9]*))$}
135  } else { ;# positive sign
136  set pattern {^[+]?(()|0|([1-9][0-9]*))$}
137  }
138  # Weak integer checking: allow empty string, empty sign, reject octals
139  set weakCheck [regexp $pattern $X]
140  # if weak check fails, continue with old value
141  if {! $weakCheck} {set X $oldX}
142  # Strong integer checking with range
143  set strongCheck [expr {[string is int -strict $X] && ($X >= $min) && ($X <= $max)}]
144

146  key {
147  $win configure -bg [expr {$strongCheck ? "white" : "yellow"}]
148  return $weakCheck
149  }
150  focusout {
151  if {! $strongCheck} {$win configure -bg red}

153  }

155  return 1
156  }
157  }
