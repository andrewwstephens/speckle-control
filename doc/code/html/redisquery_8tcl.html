29 proc redisReader { fh } {
30 global TELEMETRY
31  while { [gets $fh res] > -1 } {
32 ### debuglog "redis : $res"
33  if { [lindex $res 0] == "timestamp" || [string range $res 0 3] == "tcs." } {
