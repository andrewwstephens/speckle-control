116 proc picoCommand { axis cmd } {
117 global PICOS
118  debuglog "Commanding $axis picomotor - $cmd"
119  if { $PICOS(sim) } {
120  set rec "SIM $axis $cmd"
121  set ctype [lindex $cmd 0]
122  if { $ctype == "POS" } {
123  set PICOS($axis,position) [lindex [split $cmd "= "] 2]
124  }
125  if { $ctype == "REL" } {
126  set delta [lindex [split $cmd "= "] 2]
127  set PICOS($axis,position) [expr $PICOS($axis,position) + $delta]
128  }
129  set PICOS($axis,current) $PICOS($axis,position)
130  } else {
131  debuglog "Sending PICO command $axis $cmd"
132  puts $PICOS(handle) "$PICOS($axis) $cmd"
133  after 100
134  gets $PICOS(handle) rec
135  }
136  return $rec
137 }
