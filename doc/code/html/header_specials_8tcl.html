154 proc catchdms { value } {
155  set sign ""
156  if { $value < 0.0 } {set sign "-"}
157  set value [expr abs($value)]
158  set d [expr int($value)]
159  set m [expr int(($value-$d)*60.)]
160  set s [format %6.3f [expr ($value-$d-$m/60.)*3600.]]
161  set s1 [lindex [split $s .] 0]
162  set s2 [lindex [split $s .] 1]
163  return "string $sign[format %2.2d $d]:[format %2.2d $m]:[format %2.2d $s1].[format %s $s2]"
164 }
