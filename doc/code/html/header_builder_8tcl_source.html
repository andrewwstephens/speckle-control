19 ## Documented proc \c loadstreamdefs .
20 # \param[in] from File to read stream definitions from
21 #
22 #
23 # Globals :\n
24 # TELEMETRY - Array of telemetry items for headers and database usage\n
25 # STREAM - Array of telemetry stream names\n
26 # PDEBUG - Debug verbosity\n
27 # FITSKEY - Array of FITS keywords\n
28 # FITSTXT - Array of FITS header descriptions
29 #
30 proc loadstreamdefs { {from telem.conf} } {
31 global TELEMETRY STREAMS PDEBUG FITSKEY FITSTXT
32  set fin [open $from r]
33  while { [gets $fin rec] > -1 } {
34  if { [string trim $rec] != "" } {
35  if { [string range $rec 0 0] != "#" } {
36  if { [lindex $rec 0] == "stream" || [lindex $rec 0] == "category"} {
37  lappend STREAMS [lindex $rec 1]
38  set cstream [lindex $rec 1]
39  if { $PDEBUG } {debuglog "Definition of $cstream"}
40  } else {
41  set TELEMETRY($cstream.[lindex $rec 0]) "NA"
42  set FITSKEY($cstream.[lindex $rec 0]) [lindex $rec 1]
43  set FITSTXT($cstream.[lindex $rec 0]) [lrange $rec 2 end]
44  if { $PDEBUG } {debuglog " parameter $cstream.[lindex $rec 0]"}
45  }
46  }
47  }
48  }
49  close $fin
50 }

86 ## Documented proc \c fillheader .
87 # \param[in] args Optional arguments , scope-instrument identifier
88 #
89 # Create a FITS header structure with all the required keywords
90 #
91 # Globals :\n
92 # TELEMETRY - Array of telemetry items for headers and database usage\n
93 # SEQNUM - Sequence number\n
94 # CACHETELEMETRY - Cache of telemetry items\n
95 # PDEBUG - Debug verbosity\n
96 # HEADERS - Names of header types\n
97 # FITSKEY - Array of FITS keywords\n
98 # FITSTXT - Array of FITS header descriptions\n
99 # FROMSTARTEXP - Array of values to use from exposure start
100 # ANDOR_ARM - Instrument arm, red or blue
101 #
102 proc fillheader { args } {
103 global TELEMETRY PDEBUG HEADERS FITSKEY FITSTXT SEQNUM SCOPE
104 global FROMSTARTEXP CACHETELEMETRY ANDOR_ARM
105  set fhead ""
106  speckleTelemetryUpdate
107  updateAndorTelemetry $ANDOR_ARM
108  set type wiyn-speckle
109  if { $SCOPE(telescope) == "GEMINI" } {set type gemini-speckle}
110  set fhead "[fitshdrrecord HDR_REV string {3.00 18-Feb-2008} Header-Rev]\n"
111  foreach i $HEADERS($type) {
112  if { $PDEBUG > 1 } {debuglog "processing $i"}
113  if { [info exists FITSKEY($i)] } {
114  set key $FITSKEY($i)
115  set text $FITSTXT($i)
116  } else {
117  set key [string toupper [lindex [split $i .] 2]]
118  set text "from $i"
119  }
120  if { $key == "PROC" } {
121  set cmd "[lindex $text 0] \"$TELEMETRY($i)\""
122  set key [lindex $text 1]
123  set text [lrange $text 2 end]
124  set type string
125  catch {set type "$TELEMETRY($i,t)"}
126  set value "$TELEMETRY($i)"
127  if { [info exists FROMSTARTEXP($key)] } {
128  set value $CACHETELEMETRY($i)
129  }
130  if { $value == "Attribute" && $key != "FOCUS" } {
131  set type string
132  set value "Not available"
133  } else {
134  set parse [eval $cmd]
135  set value [lrange $parse 1 end]
136  set type [lindex $parse 0]
137  }
138  } else {
139 ### set current [$TOMPG info $i]
140 ### for testing
141  set value NA
142  catch {set value "$TELEMETRY($i)"}
143 # if { [info exists FROMSTARTEXP($key)] } {
144 # set value $CACHETELEMETRY($i)
145 # }
146  set type string
147  if { $SCOPE(telescope) == "wiyn" } {
148  catch {set type "$TELEMETRY($i,t)"}
149  if { $value == "Attribute" } {
150  set type string
151  set value "Not available"
152  }
153  }
154  }
155  set new [fitshdrrecord $key $type $value $text]
156  if { [string trim [lindex [split $new "/"] 0]] != "" } {
157  if { $PDEBUG > 1 } {debuglog "HEADER->$new"}
158  set fhead "$fhead$new\n"
159  } else {
160  debuglog "No value for $key"
161  }
162  }
163  set fhead "$fhead[fitshdrrecord OBSID string [obsid] Observation-ID]\n"
164  set fhead "$fhead[fitshdrrecord SPKLESEQ integer $SEQNUM SPECKLENFO-sequence]\n"
165  set fhead "$fhead[fitshdrrecord SPKLEDAT string [lrange [exec date] 1 3] SPECKLENFO-timestamp]"
166  if { [string trim [lindex $args 1]] != "" } {
167  set fdbg [open [lindex $args 1] w]
168  puts $fdbg "$fhead"
169  close $fdbg
170  set fhead "OK"
171  }
172  incr SEQNUM 1
173  return "$fhead"
174 }

249 ## Documented proc \c fitshdrrecord .
250 # \param[in] key A FITS header keyword
251 # \param[in] type Type of the information, integer, float, double, string
252 # \param[in] value The data value
253 # \param[in] text The text description of the item
254 #
255 # Format a FITS header record
256 #
257 proc fitshdrrecord { key type value text } {
258  set record ""
259  set v1 [lindex $value 0]
260  set fmt 18.4f
261  if { [llength [split $type _]] > 1 } {
262  set fmt [lindex [split $type _] 1]
263  set type [lindex [split $type _] 0]
264  }
265  if { $value == "NA" } {set type string}
266  switch $type {
267  string {
268  set record "[format %-8s $key]= [format %18s '[string trim $value]']"
269  }
270  integer {
271  set record "[format %-8s $key]= [format %19d [expr int($value)]]"
272  }
273  boolean {
274  set record "[format %-8s $key]= [format %19s $value]"
275  }
276  float {
277  if { [expr abs($value)] < 10000 } {set fmt 18.6f}
278  set record [string toupper "[format %-8s $key]= [format %$fmt $v1]"]
279  }
280  double {
281  if { [expr abs($value)] < 10000 } {set fmt 18.6f}
282  if { [expr abs($value)] < 1 } {set fmt 18.12f}
283  set record [string toupper "[format %-8s $key]= [format %$fmt $v1]"]
284  }
285  }
286  set record "$record /[format %-48s $text]"
287 # if { $TOMPG == "wiyn" } {
288 # set record "|TELESCOPE seqstar $record"
289 # }
290  return "$record"
291 }

294 ## Documented proc \c appendHeader .
295 # \param[in] imgname Name of FITS file
296 #
297 # Add the FITS header information to a file
298 #
299 # Globals :\n
300 # TELEMETRY - Array of telemetry items for headers and database usage\n
301 # SCOPE - Array of telescope parameters\n
302 # env - Environment variables\n
303 # SPECKLEHDRLOG - Cache of header details
304 #
305 proc appendHeader { imgname } {
306 global SPECKLEHDRLOG SCOPE env TELEMETRY
307  set hdr [fillheader $env(TELESCOPE)-$SCOPE(instrument)]
308  puts $SPECKLEHDRLOG "$hdr"
309  set fid [fits open $imgname]
310  foreach rec [split $hdr \n] {
311  $fid put keyword "$rec"
312  }
313  if { [info proc headerStandard] == "headerStandard" } {
314  headerStandard $fid
315  }
316  if { [info proc headerGeometry] == "headerGeometry" } {
317  headerGeometry $fid
318  }
319  if { [info proc headerAstrometry] == "headerAstrometry" } {
320  headerAstrometry $fid $TELEMETRY(tcs.telescope.ra) $TELEMETRY(tcs.telescope.dec)
321  }
322  if { [info proc header$SCOPE(instrument)] == "header$SCOPE(instrument)" } {
323  eval {header$SCOPE(instrument) $fid}
324  }
325  if { [info proc headerComments] == "headerComments" } {
326  fits close $fid
327  set fid [fits open $imgname]
328  headerComments $fid
329  }
330  fits close $fid
331 }

344 ## Documented proc \c calculateXSEC .
345 # \param[in] name Geometry section name
346 #
347 # Calculate image geometry header data
348 #
349 #
350 # Globals :\n
351 # SCOPE - Array of telescope parameters\n
352 # IMGMETA - Image geometry metadata
353 #
354 proc calculateXSEC { name } {
355 global IMGMETA SCOPE
356  set imgcols [expr int($IMGMETA(imageCols,postvalue))]
357  set imgrows [expr int($IMGMETA(imageRows,postvalue))]
358  set scicols [expr int($IMGMETA(sciCols,postvalue))]
359  set scirows [expr int($IMGMETA(sciRows,postvalue))]
360  set oricols [expr int($IMGMETA(pxlCols,postvalue)/$SCOPE(colBin))]
361  set orirows [expr int($IMGMETA(pxlRows,postvalue)/$SCOPE(rowBin))]
362  switch $name {
363  DATASEC {
364  set sec "\[1:$scicols,1:$scirows\]"
365  }
366  ORIGSEC {
367  set sec "\[1:$oricols,1:$orirows\]"
368  }
369  CCDSEC {
370 #must change to scistart:sciend for subrasters
371 # maybe xStart:scicols,yStart:scirows
372  set sec "\[1:$scicols,1:$scirows\]"
373  }
374  TRIMSEC {
375  set sec "\[1:$scicols,1:$scirows\]"
376  }
377  BIASSEC {
378  set startx [expr int($IMGMETA(sciCols,postvalue)+$IMGMETA(xPreScan,postvalue)+3)]
379  set endx [expr int($IMGMETA(imageCols,postvalue)-3)]
380  set sec "\[$startx:$endx,1:$scirows\]"
381  }
382  }
383  return "$sec"
384 }
