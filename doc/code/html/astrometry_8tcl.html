114 proc dms_to_radians { dms } {
115 global PI
116  set t [string trim $dms "+ "]
117  set s 1
118  if { [string range $t 0 0] == "-" } {
119  set s -1
120  set dms [string trim $dms "-"]
121  }
122  set f [split $dms ":"]
123  set r [expr $s * ([scan [lindex $f 0] %d] + [scan [lindex $f 1] %d]/60.0 + [scan [lindex $f 2] %d]/3600.0 )/180. * $PI]
